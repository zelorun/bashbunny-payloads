#!/bin/bash
# --------------------------------------------------------------------------------
# 
# --------------------------------------------------------------------------------

LED SETUP

mount -o sync /dev/nandf /root/udisk
chown root:root /root/udisk -R
chmod 777 /root/udisk -R

LOOTBASE=/root/udisk/loot
mkdir -p $LOOTBASE


# --------------------------------------------------------------------------------
# 
# --------------------------------------------------------------------------------

LED SETUP

GET SWITCH_POSITION

ATTACKMODE STORAGE HID


# --------------------------------------------------------------------------------
# 
# --------------------------------------------------------------------------------

LED STAGE1

if [ -e "$LOOTBASE/NOP.txt" ]; then
	LED FINISH
	exit 0
fi

QUACK GUI r
QUACK DELAY 250
QUACK STRING "cmd /c type nul>D:/loot/WINDOWS.txt"
QUACK DELAY 250
QUACK ENTER


# --------------------------------------------------------------------------------
# 
# --------------------------------------------------------------------------------

LED CLEANUP

sync


# --------------------------------------------------------------------------------
# 
# --------------------------------------------------------------------------------

LED FINISH
